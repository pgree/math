314  explicit matrix_cl(Mat&& A,
315  matrix_cl_view partial_view = matrix_cl_view::Entire)
316  : rows_(A.rows()), cols_(A.cols()), view_(partial_view) {
317  using Mat_type = std::decay_t<decltype(A.eval())>;
318  if (size() == 0) {
319  return;
320  }
321  if (is_eigen_matrix_or_array<Mat>::value

324  } else {

326  try {

331  // goes out of scope.
332  e.setCallback(CL_COMPLETE, &delete_it<Mat_type>, A_heap);
