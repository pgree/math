116  double t) const {
117  using std::vector;
118 
119  try {
120  start_nested();
121 
122  vector<var> y_vars;
123  y_vars.reserve(N_);
124  for (std::size_t i = 0; i < N_; ++i)
125  y_vars.emplace_back(new vari(z[i], false));
126 
127  vector<var> dy_dt_vars = f_(t, y_vars, theta_nochain_, x_, x_int_, msgs_);
128 
129  check_size_match("coupled_ode_system", "dz_dt", dy_dt_vars.size(),
130  "states", N_);
131 
132  for (size_t i = 0; i < N_; i++) {
133  dz_dt[i] = dy_dt_vars[i].val();
134  dy_dt_vars[i].grad();
135 
136  for (size_t j = 0; j < M_; j++) {
137  // orders derivatives by equation (i.e. if there are 2 eqns
138  // (y1, y2) and 2 parameters (a, b), dy_dt will be ordered as:
139  // dy1_dt, dy2_dt, dy1_da, dy2_da, dy1_db, dy2_db
140  double temp_deriv = theta_nochain_[j].adj();
141  const size_t offset = N_ + N_ * j;
142  for (size_t k = 0; k < N_; k++) {
143  temp_deriv += z[offset + k] * y_vars[k].adj();
144  }
145 
146  dz_dt[offset + i] = temp_deriv;
147  }
148 
149  set_zero_all_adjoints_nested();
150  // Parameters stored on the outer (non-nested) nochain stack are not
151  // reset to zero by the last call. This is done as a separate step here.
152  // See efficiency note above on template specalization for more details
153  // on this.
154  for (size_t j = 0; j < M_; ++j) {
155  theta_nochain_[j].vi_->set_zero_adjoint();
156  }
157  }
158  } catch (const std::exception& e) {
159  recover_memory_nested();
160  throw;
161  }
