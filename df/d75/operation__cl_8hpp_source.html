136  inline kernel_parts get_kernel_parts(
137  std::set<const operation_cl_base*>& generated, name_generator& name_gen,
138  const std::string& i, const std::string& j) const {
139  kernel_parts res{};
140  if (generated.count(this) == 0) {
141  this->var_name = name_gen.generate();
142  generated.insert(this);
143  std::string i_arg = i;
144  std::string j_arg = j;
145  derived().modify_argument_indices(i_arg, j_arg);
146  std::array<kernel_parts, N> args_parts = index_apply<N>([&](auto... Is) {
147  return std::array<kernel_parts, N>{
148  std::get<Is>(arguments_)
149  .get_kernel_parts(generated, name_gen, i_arg, j_arg)...};
150  });
151  res.body
152  = std::accumulate(args_parts.begin(), args_parts.end(), std::string(),
153  [](const std::string& a, const kernel_parts& b) {
154  return a + b.body;
155  });
156  res.args
157  = std::accumulate(args_parts.begin(), args_parts.end(), std::string(),
158  [](const std::string& a, const kernel_parts& b) {
159  return a + b.args;
160  });
161  kernel_parts my_part = index_apply<N>([&](auto... Is) {
162  return this->derived().generate(i, j,
163  std::get<Is>(arguments_).var_name...);
164  });
165  res.body = my_part.body_prefix + res.body + my_part.body;
166  res.args += my_part.args;
167  }
168  return res;
169  }
